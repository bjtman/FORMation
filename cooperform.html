//add global variables here: var synth = T("OscGen", {wave:"saw", mul:0.25}).play(); var keydict = T("ndict.key"); var midicps = T("midicps"); var WaveStrand = function(wave, x) { var t = 0; var generate = false; var self = {}; var pinHeights = []; for (var y = 0; y < xForm.y_size; y++) { pinHeights.push(0); } self.step = function() { var next_value = 0; if (generate) { next_value = wave(t); } pinHeights.pop(); pinHeights.unshift(next_value); for (var y = 0; y < xForm.y_size; y++) { xForm.setPinHeight(x, xForm.y_size - y, pinHeights[y]) } t++; } self.setGeneration = function(g) { generate = g; }; return self; } var wavestrands = [ WaveStrand(function(t) { return 0.5 + 0.5 * Math.sin(t); }, 4), WaveStrand(function(t) { return 0.5 + 0.5 * Math.sin(15*t); }, 8), WaveStrand(function(t) { return 0.5 + 0.5 * Math.cos(50*t); }, 12), WaveStrand(function(t) { return 0.5 + 0.5 * Math.cos(15*t); }, 16), WaveStrand(function(t) { return 0.5 + 0.5 * Math.cos(t); }, 20) ] var wave_map = { 81: wavestrands[0], 87: wavestrands[1], 69: wavestrands[2], 82: wavestrands[3], 84: wavestrands[4] } var body = document.getElementsByTagName("body")[0]; body.onkeydown = function(e) { if (e.keyCode in wave_map) { wave_map[e.keyCode].setGeneration(true); if(e.keyCode == 81) { synth.noteOnWithFreq(440, 100); } else if(e.keyCode == 87) { synth.noteOnWithFreq(500, 100); } else if(e.keyCode == 69) { synth.noteOnWithFreq(600, 100); } else if(e.keyCode == 82) { synth.noteOnWithFreq(700, 100); } else if(e.keyCode == 84) { synth.noteOnWithFreq(800, 100); } } // timbre.js play a note }; body.onkeyup = function(e) { if (e.keyCode in wave_map) { wave_map[e.keyCode].setGeneration(false); } // timbre.js stop playing note }; //this function will be exec
